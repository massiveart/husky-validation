Validation Options

Name Default Description
submitEvent true avoid submit if not valid
element.trigger focusout default validate trigger for each element
element.addclasses true add error and success classes for each element
{% highlight javascript %} define(['js/validation'], function(Validation) { var validation = new Validation($('#contact-form'), { submitEvent: false, element: { trigger: 'keyup', addclasses: false } }); }); {% endhighlight %}
